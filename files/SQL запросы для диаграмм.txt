SELECT Сотрудники.[Табельный номер], Заказ.Дата, Count(Заказ.[Код заказа]) AS [Count-Код заказа]
FROM Заказ INNER JOIN Сотрудники ON Заказ.[Табельный номер] = Сотрудники.[Табельный номер]
GROUP BY Сотрудники.[Табельный номер], Заказ.Дата
HAVING (((Заказ.Дата)>=[Введите дату с которой начать подсчёт (включительно)]));

диаграм выручки

SELECT Заказ.Дата, Sum((Hour([Время ухода из кафе])-Hour([Время прихода в кафе]))*150) AS Выручка
FROM Заказ
GROUP BY Заказ.Дата
HAVING (((Заказ.Дата)>=[Введите дату с которой начать подсчёт (включительно)]));
